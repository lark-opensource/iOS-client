public struct ByteViewThemeColor: UDBizColor {

    public init() {}

    public func getValueByToken(_ token: String) -> UIColor? {
        let vcToken = UDColor.Name(token)
        switch vcToken {
        case .vcTokenVCBgBody: return UIColor.ud.vcTokenVCBgBody
        case .vcTokenVCBgVideoCall: return UIColor.ud.vcTokenVCBgVideoCall
        case .vcTokenVCBgMobileShareBar: return UIColor.ud.vcTokenVCBgMobileShareBar
        case .vcTokenVCBgFloat: return UIColor.ud.vcTokenVCBgFloat
        case .vcTokenVCBgCamOff: return UIColor.ud.vcTokenVCBgCamOff
        case .vcTokenVCBtnBgOpacity: return UIColor.ud.vcTokenVCBtnBgOpacity
        case .vcTokenVCBtnTextSelected: return UIColor.ud.vcTokenVCBtnTextSelected
        case .vcTokenVCBtnFillSelected: return UIColor.ud.vcTokenVCBtnFillSelected
        case .vcTokenVCBtnIconSelected: return UIColor.ud.vcTokenVCBtnIconSelected
        case .vcTokenVCBgFloatVideo: return UIColor.ud.vcTokenVCBgFloatVideo
        case .vcTokenVCBtnBgFloat: return UIColor.ud.vcTokenVCBtnBgFloat
        case .vcTokenVCBtnBgFloatHover: return UIColor.ud.vcTokenVCBtnBgFloatHover
        case .vcTokenVCBtnBgFloatPressed: return UIColor.ud.vcTokenVCBtnBgFloatPressed
        case .vcTokenVCBgSubtitle: return UIColor.ud.vcTokenVCBgSubtitle
        case .vcTokenVCBgCardEnded: return UIColor.ud.vcTokenVCBgCardEnded
        case .vcTokenVCFillHover: return UIColor.ud.vcTokenVCFillHover
        case .vcTokenVCShadowSm: return UIColor.ud.vcTokenVCShadowSm
        case .vcTokenVCBgCardLeft: return UIColor.ud.vcTokenVCBgCardLeft
        case .vcTokenVCBgCardRight: return UIColor.ud.vcTokenVCBgCardRight
        case .vcTokenVCannotateBgRed: return UIColor.ud.vcTokenVCannotateBgRed
        case .vcTokenVCannotateBgYellow: return UIColor.ud.vcTokenVCannotateBgYellow
        case .vcTokenVCannotateBgGreen: return UIColor.ud.vcTokenVCannotateBgGreen
        case .vcTokenVCannotateBgBlue: return UIColor.ud.vcTokenVCannotateBgBlue
        case .vcTokenVCannotateBgPurple: return UIColor.ud.vcTokenVCannotateBgPurple
        case .vcTokenVCBtnBgNewMeeting: return UIColor.ud.vcTokenVCBtnBgNewMeeting
        case .vcTokenVCBtnBgSchedule: return UIColor.ud.vcTokenVCBtnBgSchedule
        case .vcTokenVCBtnBgShareScreen: return UIColor.ud.vcTokenVCBtnBgShareScreen
        case .vcTokenVCBtnBgMinutes: return UIColor.ud.vcTokenVCBtnBgMinutes
        case .vcTokenVCBtnIconNewMeeting: return UIColor.ud.vcTokenVCBtnIconNewMeeting
        case .vcTokenVCBtnIconSchedule: return UIColor.ud.vcTokenVCBtnIconSchedule
        case .vcTokenVCBtnIconShareScreen: return UIColor.ud.vcTokenVCBtnIconShareScreen
        case .vcTokenVCBtnIconMinutes: return UIColor.ud.vcTokenVCBtnIconMinutes
        case .vcTokenVCListIconBgOngoing: return UIColor.ud.vcTokenVCListIconBgOngoing
        case .vcTokenVCListIconBgOngoingHover: return UIColor.ud.vcTokenVCListIconBgOngoingHover
        case .vcTokenVCListIconBgUpcoming: return UIColor.ud.vcTokenVCListIconBgUpcoming
        case .vcTokenVCListIconBgUpcomingHover: return UIColor.ud.vcTokenVCListIconBgUpcomingHover

        creatThemeColor
        default: return nil
        }
    }
}
